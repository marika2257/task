task.addEventListener("click", () => { task.remove(); showMessage("タスク完了、お疲れさま。"); addExp(30); // ← ここ追加！ saveTasks(); }); taskList.appendChild(task); taskInput.value = ""; addExp(10); // ← ここ追加！ saveTasks(); localStorage.setItem("tasks", JSON.stringify(tasks)); } function loadTasks() { const saved = JSON.parse(localStorage.getItem("tasks") || "[]"); saved.forEach(text => { const task = document.createElement("div"); task.textContent = text; task.addEventListener("click", () => { task.remove(); showMessage("タスク完了、お疲れさま。"); addExp(30); // ← ここも必要！ saveTasks(); }); taskList.appendChild(task); }); } // ========================= // ペルソナ風 EXP / レベルシステム // ========================= let player = { level: 1, exp: 0, nextExp: 100, }; function addExp(amount) { player.exp += amount; updateExpUI(); if (player.exp >= player.nextExp) { levelUp(); } } function levelUp() { player.level++; player.exp = player.exp - player.nextExp; player.nextExp = Math.floor(player.nextExp * 1.3); showLevelUpCutin(); updateExpUI(); } function updateExpUI() { const expBar = document.getElementById("exp-bar"); const levelText = document.getElementById("level-text"); const percent = (player.exp / player.nextExp) * 100; expBar.style.width = percent + "%"; levelText.textContent = "Lv " + player.level; } function showLevelUpCutin() { const cutin = document.getElementById("levelup-cutin"); cutin.classList.add("show"); setTimeout(() => { cutin.classList.remove("show"); }, 2000); }
